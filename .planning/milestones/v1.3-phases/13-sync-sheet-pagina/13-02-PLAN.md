    ```javascript
    ;(function() {
      if (!window.CLAIMS_DATA) return;
      var badges = document.querySelectorAll('[data-claim-id]');
      for (var i = 0; i < badges.length; i++) {
        var claimId = badges[i].getAttribute('data-claim-id');
        var claim = window.CLAIMS_DATA[claimId];
        if (!claim || !claim.source_reference) continue;
        var tooltip = badges[i].querySelector('.bcn-tooltip');
        if (!tooltip) continue;
        var text = claim.source_reference;
        if (claim.extracto_verbatim) {
          // Extract relevant short quote (first 120 chars or up to first period after 50 chars)
          var excerpt = claim.extracto_verbatim;
          if (claim.inciso) {
            // Try to find the specific inciso section
            var incisoPattern = claim.inciso + '°';
            var idx = excerpt.indexOf(incisoPattern);
            if (idx === -1) { incisoPattern = claim.inciso + '\u00b0'; idx = excerpt.indexOf(incisoPattern); }
            if (idx !== -1) {
              excerpt = excerpt.substring(idx);
              var nextNum = excerpt.indexOf((parseInt(claim.inciso) + 1) + '°');
              if (nextNum === -1) nextNum = excerpt.indexOf((parseInt(claim.inciso) + 1) + '\u00b0');
              if (nextNum > 0) excerpt = excerpt.substring(0, nextNum).trim();
            }
          }
          if (excerpt.length > 150) excerpt = excerpt.substring(0, 147) + '...';
          text += ': \u00ab' + excerpt + '\u00bb';
        }
        tooltip.textContent = text;
      }
    })();
    ```

    ```javascript
    /* claims-tooltips.js — Popula tooltips BCN desde CLAIMS_DATA

      'use strict';
      document.addEventListener('DOMContentLoaded', function() {
        if (!window.CLAIMS_DATA) return;
        var badges = document.querySelectorAll('[data-claim-id]');
        for (var i = 0; i < badges.length; i++) {
          var claimId = badges[i].getAttribute('data-claim-id');
          var claim = window.CLAIMS_DATA[claimId];
          if (!claim || !claim.source_reference) continue;
          var tooltip = badges[i].querySelector('.bcn-tooltip');
          if (!tooltip) continue;
          var sourceName = claim.source_name || '';
          var text = claim.source_reference;
          if (sourceName) text += ', ' + sourceName;
          if (claim.extracto_verbatim) {
            var excerpt = claim.extracto_verbatim;
            if (excerpt.length > 200) excerpt = excerpt.substring(0, 197) + '...';
            text += ': \u00ab' + excerpt + '\u00bb';
          }
          tooltip.textContent = text;
        }
      });
    })();
    ```
